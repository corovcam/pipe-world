56 private static CellWalls[,] Backtrack(CellWalls[,] cells, int width, int height)
57 {
58 var posStack = new Stack<Position>();
59 var position = new Position { X = 0, Y = 0 };
60
61 // Mark first position as visited
62 cells[position.X, position.Y] |= CellWalls.VISITED;
63 posStack.Push(position);
64
65 // When there are unvisited Cells, continue
66 while (posStack.Count > 0)
67 {
68 Position current = posStack.Pop();
69 // Look for nearby unvisited Cells to continue
70 var adjacents = GetUnvisitedAdjacentCells(current, cells, width, height);
71
72 if (adjacents.Count > 0)
73 {
74 posStack.Push(current);
75
76 // Choose a random adjacent Cell to continue (to simulate randomicity of a maze)
77 int randIndex = UnityEngine.Random.Range(0, adjacents.Count);
78 AdjacentCell randomAdjacent = adjacents[randIndex];
79
80 Position randAdjPos = randomAdjacent.Location;
81 // Remove the CommonWall (of the chosen random Adjacent) from the current cell
82 cells[current.X, current.Y] &= ~randomAdjacent.CommonWall;
83 // Remove the opposite wall from the random chosen = the wall removed from the current cell
84 cells[randAdjPos.X, randAdjPos.Y] &= ~GetOppositeWall(randomAdjacent.CommonWall);
85 // Mark the new cell as visited
86 cells[randAdjPos.X, randAdjPos.Y] |= CellWalls.VISITED;
87
88 posStack.Push(randAdjPos);
89 }
90 }
91
92 return cells;
93 }

175 public static CellWalls[,] GenerateMaze(int width, int height)
176 {
177 CellWalls[,] maze = new CellWalls[width, height];
178
179 // All cells in the maze are initialized with all walls present
180 CellWalls initialCell = CellWalls.RIGHT | CellWalls.LEFT | CellWalls.UP | CellWalls.DOWN;
181 for (int i = 0; i < width; ++i)
182 {
183 for (int j = 0; j < height; ++j)
184 {
185 maze[i, j] = initialCell; // 1111
186 }
187 }
188
189 // The algorithm will remove some walls to create a maze
190 return Backtrack(maze, width, height);
191 }
