102 void GenerateNewGrid()
103 {
104 // Choose the BackTile Sprite at random
105 Sprite chosenBackTileSprite = backTileSprites[Random.Range(0, backTileSprites.Count)];
106 // For every BackTile position create a new GO, configure it, set its Transform and store it in tileObjects
107 for (int y = boardSize - 1; y >= 0; y--)
108 {
109 for (int x = 0; x < boardSize; x++)
110 {
111 GameObject temp = Instantiate(backTilePrefab);
112 temp.GetComponent<SpriteRenderer>().sprite = chosenBackTileSprite;
113 Vector2 tileTransform = new Vector2(x, y);
114 temp.transform.position = tileTransform;
115 temp.name = string.Format("({0}, {1})", x, y);
116 temp.transform.parent = gameObject.transform;
117
118 /* Sets the layer of tiles to "Ignore Raycast" to Raycast Pipes only.
119 Prevents various unexpected errors when using built-in Physics2D engine. */
120 if (LevelData.IsFreeWorldMode)
121 {
122 temp.layer = LayerMask.NameToLayer("Ignore Raycast");
123 var collider = temp.AddComponent<BoxCollider2D>();
124 collider.isTrigger = true;
125 collider.size = new Vector2(1, 1);
126 }
127
128 tileObjects[x, y] = temp;
129 }
130 }
131 // Position the entire grid in the middle of the Camera ViewPort
132 gameObject.transform.position = new Vector2(-(float)(boardSize / 2.0 - 0.5), -(float)(boardSize / 2.0 - 0.5));

138 void GenerateLevel()
139 {
140 Pipe[,] pipes;
141 if (LevelData.IsArcadeMode)
142 pipes = LevelData.GetRandomPuzzle(boardSize, boardSize);
143 else
144 pipes = LevelData.ReadInputLevelData();
145
146 // Offset is used to transform the Unity's default coord system from the lower-left corner
147 // to a more practical upper-left corner start position (0,0)
148 int offset = boardSize - 1;
149 List<Pipe> pipesList = new List<Pipe>();
150 // If the level is in free world mode, add all the pipes to a list. This is used to randomly select pipes without replacements.
151 if (LevelData.IsFreeWorldMode)
152 {
153 foreach (var pipe in pipes)
154 pipesList.Add(pipe);
155 }
156
157 // For each row
158 for (int y = 0; y < boardSize; y++)
159 {
160 // For each column
161 for (int x = 0; x < boardSize; x++)
162 {
163 Pipe pipe;
164 if (LevelData.IsFreeWorldMode)
165 {
166 // If the level is in free world mode, then we want to randomly select a pipe without replacement, unless it's a start or end pipe.
167 if (LevelData.Starts.Select(kv => kv.Value).Any(v => v.Contains(new Position(x, y + offset))) ||
168 LevelData.Ends.Select(kv => kv.Value).Any(v => v.Contains(new Position(x, y + offset))))
169 {
170 pipe = pipes[x, y];
171 pipesList.Remove(pipe);
172 }
173 else
174 {
175 int randIndex = Random.Range(0, pipesList.Count);
176 pipe = pipesList[randIndex];
177 pipesList.RemoveAt(randIndex);
178 }
179 }
180 else
181 {
182 pipe = pipes[x, y];
183 }
184
185 // Instantiate the pipe prefab.
186 GameObject pipePrefab = pipe.Liquid == Liquid.Water ? waterPipePrefabs[(int)pipe.Type] : lavaPipePrefabs[(int)pipe.Type];
187 GameObject pipeGO = Instantiate(pipePrefab);
188
189 // Add the grid pipe component if the level is in free world mode.
190 if (LevelData.IsFreeWorldMode)
191 {
192 pipeGO.AddComponent<GridPipe>();
193 var rb = pipeGO.AddComponent<Rigidbody2D>();
194 rb.bodyType = RigidbodyType2D.Kinematic;
195 }
196
197 // Set the pipe type on the pipe handler.
198 pipeGO.GetComponent<PipeHandler>().pipeType = pipe;
199 PositionPipeGO(tileObjects[x, y + offset], pipeGO);
200 }
201 offset -= 2;
202 }
203 }

270 void SetStartEndPipeSprites(bool iterateStarts)
271 {
272 // Iterate through either the starting or ending positions of liquid pipes in a game level
273 // and update their sprites based on their liquid type and whether they are filled or not.
274 // If the game is in free world mode, the GridPipe component of the start/end pipes is also destroyed
275 // (to prevent dragging events).
276 var iterable = iterateStarts ? LevelData.Starts : LevelData.Ends;
277
278 foreach (Liquid liq in iterable.Keys)
279 {
280 foreach (Position pos in iterable[liq])
281 {
282 var pipe = LevelData.GamePieces[pos.X, pos.Y];
283 Pipe pipeType = pipe.pipeType;
284 Sprite chosenSprite;
285
286 if (iterateStarts)
287 {
288 chosenSprite = pipeType.Liquid == Liquid.Water ?
289 filledBlueGreenPipeSprites[(int)pipeType.Type] : filledRedGreyPipeSprites[(int)pipeType.Type];
290 }
291 else
292 {
293 chosenSprite = pipeType.Liquid == Liquid.Water ?
294 blueGreenPipeSprites[(int)pipeType.Type] : redGreyPipeSprites[(int)pipeType.Type];
295 }
296
297 pipe.GetComponent<SpriteRenderer>().sprite = chosenSprite;
298
299 if (LevelData.IsFreeWorldMode)
300 {
301 Destroy(pipe.GetComponent<GridPipe>());
302 }
303 }
304 }

311 public void ResetLevel()
312 {
313 // If Free World mode is active, do nothing and return
314 if (LevelData.IsFreeWorldMode)
315 return;
316
317 // Reset each pipe to its default sprite variant (with or without a liquid)
318 foreach (var pipe in LevelData.GamePieces)
319 {
320 Pipe pipeType = pipe.pipeType;
321 if (pipeType.Type != PipeType.EMPTY)
322 {
323 var chosenSprite = pipeType.Liquid == Liquid.Water ?
324 greenPipeSprites[(int)pipeType.Type] : greyPipeSprites[(int)pipeType.Type];
325 pipe.GetComponent<SpriteRenderer>().sprite = chosenSprite;
326 }
327 }
328
329 // Set the sprites for the StartPipe and EndPipe
330 SetStartEndPipeSprites(iterateStarts: true);
331 SetStartEndPipeSprites(iterateStarts: false);
332
333 // Set the Active Tile to the StartPipe
334 SetActiveTile(LevelData.GamePieces[LevelData.defaultStart.Value.X, LevelData.defaultStart.Value.Y].gameObject);
335
336 // Shuffle the GamePieces
337 StartCoroutine(Shuffle());
338 }
