140 IEnumerator Flow(Position startPos, Liquid flowLiquid, int flowIndex)
141 {
142 PipeHandler startPipe = LevelData.GamePieces[startPos.X, startPos.Y];
143 PipeHandler previousPipe = startPipe;
144 distances[startPipe.location] = 0;
145 visited.Add(startPipe);
146 queue.Enqueue(startPipe);
147
148 // Loops the queue until all connected pipes have been visited
149 while (queue.Count != 0)
150 {
151 var current = queue.Dequeue();
152
153 // If the previous pipe is 1 level below current, then the whole level
154 // has been visited and wait for next wave (used for wave animation)
155 if (distances[previousPipe.location] < distances[current.location])
156 if (flowLiquid == Liquid.Water)
157 yield return new WaitForSeconds(0.5f);
158 else
159 yield return new WaitForSeconds(1f);
160
161 Pipe currentType = current.pipeType;
162 // Set filled Blue-Green/Red-Grey sprites for Start/End sprites
163 if (current.location == startPipe.location || LevelData.Ends[flowLiquid].Exists(pos => pos == current.location))
164 {
165 var chosenStartEndSprite = currentType.Liquid == Liquid.Water ?
166 lh.filledBlueGreenPipeSprites[(int)currentType.Type] : lh.filledRedGreyPipeSprites[(int)currentType.Type];
167 current.GetComponent<SpriteRenderer>().sprite = chosenStartEndSprite;
168 }
169 else // Otherwise set filled green/grey sprites
170 {
171 var chosenSprite = currentType.Liquid == Liquid.Water ?
172 lh.filledGreenPipeSprites[(int)currentType.Type] : lh.filledGreyPipeSprites[(int)currentType.Type];
173 current.GetComponent<SpriteRenderer>().sprite = chosenSprite;
174 }
175
176 // Loop through each IO Dir of the Pipe
177 for (int dir = 0; dir < current.IODirs.Length; dir++)
178 {
179 if (current.IODirs[dir]) // If the IO Port is available
180 {
181 switch (dir) // Check the direction and continue there
182 {
183 case (int)Dir.UP:
184 // Check if we can move the water to the UP Pipe, check if it hasn't been already visited and
185 // also the the liquid type must be the same, otherwise the liquid doesn't move through
186 if (current.upFree && !visited.Contains(current.up) && currentType.Liquid == current.up.pipeType.Liquid)
187 {
188 previousPipe = current; // Remember the previous Pipe for wave animation
189 distances[current.up.location] = distances[current.location] + 1;
190 visited.Add(current.up);
191 queue.Enqueue(current.up);
192 }
193 break;
194 case (int)Dir.RIGHT:
195 if (current.rightFree && !visited.Contains(current.right) && currentType.Liquid == current.right.pipeType.Liquid)
196 {
197 previousPipe = current;
198 distances[current.right.location] = distances[current.location] + 1;
199 visited.Add(current.right);
200 queue.Enqueue(current.right);
201 }
202 break;
203 case (int)Dir.DOWN:
204 if (current.downFree && !visited.Contains(current.down) && currentType.Liquid == current.down.pipeType.Liquid)
205 {
206 previousPipe = current;
207 distances[current.down.location] = distances[current.location] + 1;
208 visited.Add(current.down);
209 queue.Enqueue(current.down);
210 }
211 break;
212 case (int)Dir.LEFT:
213 if (current.leftFree && !visited.Contains(current.left) && currentType.Liquid == current.left.pipeType.Liquid)
214 {
215 previousPipe = current;
216 distances[current.left.location] = distances[current.location] + 1;
217 visited.Add(current.left);
218 queue.Enqueue(current.left);
219 }
220 break;
221 }
222 }
223 }
224 }

226 flowsFinished[flowIndex] = true;
